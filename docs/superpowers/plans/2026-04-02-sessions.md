# Sessions Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add session rooms to the NHL Radio Overlay so multiple users sharing a short code receive the same stat cards and all can broadcast audio.

**Architecture:** An in-memory `SessionStore` in a new `backend/session_store.py` replaces the global `_ws_clients` set. Both WebSocket endpoints (`/ws`, `/audio`) accept a `?session=CODE` query param. A new `SessionSplash` React component gates the overlay; `useSession` owns all session state and is consumed by `App` and `MicHud`.

**Tech Stack:** FastAPI (Python), pytest, React 18, TypeScript, Vitest + Testing Library, Tailwind CSS

```python
"""In-memory session store for the NHL Radio Overlay."""
from __future__ import annotations

import random
import time
from dataclasses import dataclass, field
from typing import TYPE_CHECKING

from config import SESSION_CODE_CHARS, SESSION_CODE_LENGTH

if TYPE_CHECKING:
    from fastapi import WebSocket


@dataclass
class Session:
    code: str
    ws_clients: list["WebSocket"] = field(default_factory=list)
    audio_clients: set["WebSocket"] = field(default_factory=set)
    locked: bool = False
    created_at: float = field(default_factory=time.time)


class SessionStore:
    """Thread-safe-enough (asyncio single-threaded) in-memory session registry."""

    def __init__(self) -> None:
        self._sessions: dict[str, Session] = {}

    # ------------------------------------------------------------------
    # Read
    # ------------------------------------------------------------------

    def get(self, code: str) -> Session | None:
        return self._sessions.get(code)

    def is_creator(self, code: str, ws: "WebSocket") -> bool:
        session = self._sessions.get(code)
        if session is None or not session.ws_clients:
            return False
        return session.ws_clients[0] is ws

    # ------------------------------------------------------------------
    # Write
    # ------------------------------------------------------------------

    def create(self) -> str:
        """Generate a collision-safe session code and register the session."""
        while True:
            code = "".join(random.choices(SESSION_CODE_CHARS, k=SESSION_CODE_LENGTH))
            if code not in self._sessions:
                self._sessions[code] = Session(code=code)
                return code

    def join(self, code: str, ws: "WebSocket") -> None:
        """Add *ws* to the session's overlay client list.

        Raises KeyError if the session does not exist.
        Raises PermissionError if the session is locked.
        """
        session = self._sessions.get(code)
        if session is None:
            raise KeyError(code)
        if session.locked:
            raise PermissionError("session is locked")
        session.ws_clients.append(ws)

    def leave(self, code: str, ws: "WebSocket") -> None:
        """Remove *ws* from the session. Deletes the session if it becomes empty."""
        session = self._sessions.get(code)
        if session is None:
            return
        try:
            session.ws_clients.remove(ws)
        except ValueError:
            return  # ws was not in the list — ignore
        if not session.ws_clients:
            del self._sessions[code]

    def join_audio(self, code: str, ws: "WebSocket") -> None:
        session = self._sessions.get(code)
        if session is not None:
            session.audio_clients.add(ws)

    def leave_audio(self, code: str, ws: "WebSocket") -> None:
        session = self._sessions.get(code)
        if session is not None:
            session.audio_clients.discard(ws)

    def set_locked(self, code: str, locked: bool) -> None:
        session = self._sessions.get(code)
        if session is not None:
            session.locked = locked
```

```python
# ---------------------------------------------------------------------------
# SERV-04: session-scoped broadcast helper
# ---------------------------------------------------------------------------

async def broadcast_to_session(session_code: str, payload: dict) -> None:
    """Wrap payload in the v1 envelope and send to all /ws clients in the session."""
    session = _session_store.get(session_code)
    if session is None:
        return
    message = json.dumps({"v": 1, "payload": payload})
    dead: list[WebSocket] = []

    async def _safe_send(ws: WebSocket) -> None:
        try:
            await ws.send_text(message)
        except Exception:
            dead.append(ws)

    await asyncio.gather(*(_safe_send(ws) for ws in session.ws_clients))
    for ws in dead:
        _session_store.leave(session_code, ws)


async def _send_one(ws: WebSocket, payload: dict) -> None:
    """Send a payload to a single WebSocket client."""
    try:
        await ws.send_text(json.dumps({"v": 1, "payload": payload}))
    except Exception:
        pass


# ---------------------------------------------------------------------------
# SERV-03: WS /ws — overlay clients receive broadcasted stat payloads
# ---------------------------------------------------------------------------

@app.websocket("/ws")
async def overlay_ws(websocket: WebSocket, session: str = "") -> None:
    if not session or _session_store.get(session) is None:
        await websocket.close(code=4000)
        return
    await websocket.accept()
    try:
        _session_store.join(session, websocket)
    except PermissionError:
        await _send_one(websocket, _system_payload("session_rejected", "Session is locked"))
        await websocket.close(code=4001)
        return

    is_creator = _session_store.is_creator(session, websocket)
    count = len(_session_store.get(session).ws_clients)

    await _send_one(websocket, _system_payload("connected", "Overlay connected"))
    await broadcast_to_session(session, {
        **_system_payload("session_joined", "Member joined"),
        "participant_count": count,
        "is_creator": is_creator,
    })

    try:
        while True:
            text = await websocket.receive_text()
            try:
                msg = json.loads(text)
            except json.JSONDecodeError:
                continue
            if msg.get("type") == "session_lock" and _session_store.is_creator(session, websocket):
                locked = bool(msg.get("locked", True))
                _session_store.set_locked(session, locked)
                event = "session_locked" if locked else "session_unlocked"
                await broadcast_to_session(session, {
                    **_system_payload(event, f"Session {'locked' if locked else 'unlocked'}"),
                    "locked": locked,
                })
    except WebSocketDisconnect:
        pass
    finally:
        was_creator = _session_store.is_creator(session, websocket)
        _session_store.leave(session, websocket)
        remaining = _session_store.get(session)
        if remaining:
            count = len(remaining.ws_clients)
            await broadcast_to_session(session, {
                **_system_payload("session_left", "Member left"),
                "participant_count": count,
            })
            if was_creator and remaining.ws_clients:
                new_creator = remaining.ws_clients[0]
                await _send_one(new_creator, {
                    **_system_payload("session_creator_changed", "You are now the session creator"),
                    "is_creator": True,
                })
```

```typescript
import { useCallback, useRef, useState } from 'react'
import type { MutableRefObject } from 'react'
import type { SystemPayload } from '../types/payloads'

export interface UseSessionReturn {
  sessionCode: string | null
  isCreator: boolean
  isLocked: boolean
  participantCount: number
  joinSession(code: string): Promise<void>
  createSession(): Promise<void>
  setLocked(locked: boolean): void
  handleSessionEvent(event: SystemPayload): void
  /** Populated by useOverlaySocket — used by setLocked to send WS messages */
  wsRef: MutableRefObject<WebSocket | null>
}

export function useSession(): UseSessionReturn {
  const [sessionCode, setSessionCode] = useState<string | null>(null)
  const [isCreator, setIsCreator] = useState(false)
  const [isLocked, setIsLocked] = useState(false)
  const [participantCount, setParticipantCount] = useState(0)
  const wsRef = useRef<WebSocket | null>(null)

  const createSession = useCallback(async () => {
    const resp = await fetch('/session', { method: 'POST' })
    if (!resp.ok) throw new Error('Failed to create session')
    const { code } = await resp.json()
    setSessionCode(code)
    setIsCreator(true)
    setIsLocked(false)
    setParticipantCount(0)
  }, [])

  const joinSession = useCallback(async (code: string) => {
    const resp = await fetch(`/session/${code}`)
    if (!resp.ok) throw new Error('Failed to check session')
    const data = await resp.json()
    if (!data.exists) throw new Error('not_found')
    if (data.locked) throw new Error('locked')
    setSessionCode(code)
    setIsCreator(false)
    setIsLocked(data.locked)
    setParticipantCount(data.participant_count)
  }, [])

  const setLocked = useCallback((locked: boolean) => {
    if (wsRef.current?.readyState === WebSocket.OPEN) {
      wsRef.current.send(JSON.stringify({ type: 'session_lock', locked }))
    }
  }, [])

  const handleSessionEvent = useCallback((event: SystemPayload) => {
    switch (event.event) {
      case 'session_joined':
      case 'session_left':
        if (event.participant_count !== undefined) setParticipantCount(event.participant_count)
        if (event.is_creator === true) setIsCreator(true)
        break
      case 'session_locked':
        setIsLocked(true)
        break
      case 'session_unlocked':
        setIsLocked(false)
        break
      case 'session_creator_changed':
        if (event.is_creator === true) setIsCreator(true)
        break
    }
  }, [])

  return { sessionCode, isCreator, isLocked, participantCount, joinSession, createSession, setLocked, handleSessionEvent, wsRef }
}
```

```typescript
        if (payload.type === 'system') {
          const sys = payload as SystemPayload
          if (SESSION_EVENTS.has(sys.event)) {
            onSessionEventRef.current?.(sys)
          } else {
            setSystemEvent(sys)
          }
        } else {
          setLatestPayload(payload)
        }
      } catch {
        // malformed — ignore
      }
    }

    ws.onclose = () => {
      setIsConnected(false)
      if (externalWsRefRef.current) externalWsRefRef.current.current = null
      if (!activeRef.current || !codeRef.current) return
      const delay = WS_BACKOFF_MS[Math.min(retryRef.current, WS_BACKOFF_MS.length - 1)]
      retryRef.current++
      timerRef.current = setTimeout(connect, delay)
    }
  }, [])

  useEffect(() => {
    if (!sessionCode) return
    activeRef.current = true
    connect()
    return () => {
      activeRef.current = false
      if (timerRef.current !== null) clearTimeout(timerRef.current)
      wsRef.current?.close()
    }
  }, [sessionCode, connect])

  return { latestPayload, systemEvent, isConnected }
}
```

```tsx
import { useState } from 'react'
import { SESSION_CODE_LENGTH } from '../config'
import type { UseSessionReturn } from '../hooks/useSession'

type Props = { session: UseSessionReturn }

export function SessionSplash({ session }: Props) {
  const [code, setCode] = useState('')
  const [error, setError] = useState<string | null>(null)
  const [loading, setLoading] = useState(false)

  const handleJoin = async () => {
    setError(null)
    setLoading(true)
    try {
      await session.joinSession(code)
    } catch (err) {
      const msg = err instanceof Error ? err.message : 'unknown'
      if (msg === 'locked') setError('Session is locked — ask the host to unlock it.')
      else if (msg === 'not_found') setError('Session not found. Check the code and try again.')
      else setError('Could not join session. Try again.')
    } finally {
      setLoading(false)
    }
  }

  const handleCreate = async () => {
    setError(null)
    setLoading(true)
    try {
      await session.createSession()
    } catch {
      setError('Could not create session. Try again.')
    } finally {
      setLoading(false)
    }
  }

  return (
    <div className="w-screen h-screen bg-black flex items-center justify-center">
      <div className="bg-zinc-900 border border-zinc-700 rounded-xl px-10 py-8 w-80 text-center">
        <h1 className="text-white text-xl font-bold mb-1">NHL Radio Overlay</h1>
        <p className="text-zinc-500 text-sm mb-6">Enter a session code or start a new one</p>

        <input
          type="text"
          maxLength={SESSION_CODE_LENGTH}
          value={code}
          onChange={e => setCode(e.target.value.toUpperCase())}
          placeholder="Session code"
          className="w-full bg-zinc-800 border border-zinc-600 text-white text-center text-lg tracking-widest rounded-lg px-3 py-2 mb-2 outline-none focus:border-blue-500 uppercase"
        />

        <p className="text-red-400 text-xs min-h-[16px] mb-3">
          {error ?? ''}
        </p>

        <div className="flex gap-3">
          <button
            onClick={handleJoin}
            disabled={code.length < SESSION_CODE_LENGTH || loading}
            className="flex-1 bg-blue-600 hover:bg-blue-500 disabled:opacity-40 disabled:cursor-not-allowed text-white font-semibold text-sm py-2 rounded-lg transition-colors"
          >
            Join
          </button>
          <button
            onClick={handleCreate}
            disabled={loading}
            className="flex-1 bg-green-700 hover:bg-green-600 disabled:opacity-40 disabled:cursor-not-allowed text-white font-semibold text-sm py-2 rounded-lg transition-colors"
          >
            New Session
          </button>
        </div>
      </div>
    </div>
  )
}
```
